Your first request in 3 minutes

  1. 1. Create a free account and copy your API key.
  2. 2. Run a task.
          # cURL
curl -X POST https://api.gobii.ai/api/v1/agents/ \
  -H "X-Api-Key: <YOUR_API_KEY>" \
  -H "Content-Type: application/json" \
  -d '{
    "name": "Lead Finder Agent",
    "charter": "Find new leads in my industry, and post results to the webhook every few hours",
    "schedule": "@daily",
    "preferred_contact_endpoint": "webhook"
  }'
            
        
          
const res = await fetch("https://api.gobii.ai/v1/tasks", {
  method: "POST",
  headers: {
    "X-Api-Key": process.env.GOBII_API_KEY,
    "Content-Type": "application/json"
  },
  body: JSON.stringify({
    prompt: "Fill out the onboarding form at example.com",
    callback_url: "https://example.com/webhooks/gobii"
  })
});
const json = await res.json();
console.log(json);

3. Receive results via webhook:

          
// Express.js webhook example
app.post("/webhooks/gobii", (req, res) => {
  const { task_id, status, output, error } = req.body;
  if (status === "succeeded") {
    // process output (files, text, structured data)
  } else if (status === "failed") {
    // log/alert and retry logic
  }
  res.sendStatus(200);
});
