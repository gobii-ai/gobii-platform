Your first request in 3 minutes

  1. 1. Create a free account and copy your API key.
  2. 2. Run a task.
# cURL
curl -X POST https://api.gobii.ai/v1/tasks \
  -H "Authorization: Bearer " \
  -H "Content-Type: application/json" \
  -d '{
    "prompt": "Log into example.com and export recent messages as CSV",
    "callback_url": "https://yourapp.com/webhooks/gobii"
  }'

# Node.js (fetch)
import fetch from "node-fetch";
const res = await fetch("https://api.gobii.ai/v1/tasks", {
  method: "POST",
  headers: {
    "Authorization": "Bearer " + process.env.GOBII_API_KEY,
    "Content-Type": "application/json"
  },
  body: JSON.stringify({
    prompt: "Fill out the onboarding form at example.com",
    callback_url: "https://yourapp.com/webhooks/gobii"
  })
});
const json = await res.json();
console.log(json); // { task_id, status: "queued" }

3. Receive results via webhook:

// Express.js webhook example
app.post("/webhooks/gobii", express.json(), (req, res) => {
  const { task_id, status, output, error } = req.body;
  if (status === "succeeded") {
    // process output (files, text, structured data)
  } else if (status === "failed") {
    // log/alert and retry logic
  }
  res.sendStatus(200);
});
